\newcommand{\createlength}[2] % Two args: 1) name ; 2) value
{
  \newlength{#1}     % creates the new length variable/command
  \setlength{#1}{#2} % and sets it right here
}

% Visible height of front cover. 297mm for Din A4

% Computed lengths
